# Note: UTM zones extend over 6d in longitude and a full hemisphere in
# latitude. However, the NATO system further refines the zones by
# putting them into latitude zones, to form a grid. pyproj uses the former
# approach, though. See "https://en.wikipedia.org/wiki/
# Universal_Transverse_Mercator_coordinate_system" for further details.
# In the following, we use "N" and "S" to make a distinction between
# Northern and Southern hemisphere. Don't mix this up with the grid
# nomenclature (where xxS is a cell on the Northern hemisphere...).

[docs]def utm_zone_from_gps(glon, glat):

    glon = np.atleast_1d(glon)
    glat = np.atleast_1d(glat)

    glon = (glon + 180) % 360 - 180

    zone = (np.int32(np.floor(glon + 180)) // 6 + 1).astype('<U2')
    ns = np.where(glat >= 0, 'N', 'S')

    utm = np.char.add(zone, ns).squeeze()
    if utm.size == 1:
        return str(utm)
    else:
        return utm




[docs]def epsg_from_utm_zone(utm_zone):

    match = UTM_ZONE_RE.match(utm_zone)
    if not (match and len(match.groups()) == 2):
        raise ValueError('{} is not a valid UTM zone'.format(utm_zone))

    zone, ns = match.groups()

    if ns.upper() == 'N':
        return 'EPSG:326{:02d}'.format(int(zone))
    elif ns.upper() == 'S':
        return 'EPSG:327{:02d}'.format(int(zone))



@lru_cache(maxsize=64, typed=True)
def _create_proj(sys1, sys2, zone=None, south=False):
    '''
    Helper function to create and cache pyproj.Proj instances,
    or transform functions.
    '''

    def _c(epsg_enum):
        return '+init=epsg:{:04d}'.format(epsg_enum.value)

    try:
        from pyproj import Proj, transform
    except ImportError:
        raise ImportError(
            'The "pyproj" package is necessary to use this function.'
            )

    if not (isinstance(sys1, (str, EPSG)) and isinstance(sys1, (str, EPSG))):
        raise TypeError(
            'sys1/2 must be strings or EPSG enums: {} {}'.format(sys1, sys2)
            )

    if isinstance(sys1, EPSG):
        sys1 = _c(sys1)

    if isinstance(sys1, EPSG):
        sys2 = _c(sys2)

    if sys1.upper().startswith('EPSG:') and sys2.upper().startswith('EPSG:'):

        return partial(transform, Proj(sys1), Proj(sys2))

    elif sys1 == 'UTM' and sys2 == 'WGS84':

        if not isinstance(zone, numbers.Integral):
            raise TypeError('zone must be an integer')

        _proj_str = (
            '+proj=utm +ellps=WGS84 +datum=WGS84 +units=m +no_defs '
            '+zone={:d}'.format(zone)
            )
        if south:
            _proj_str += '+south'

        return Proj(_proj_str)

    elif sys1 == 'ETRS89' and sys2 == 'WGS84':

        _proj_str = (
            '+proj=laea +lat_0=52 +lon_0=10 +x_0=4321000 +y_0=3210000 '
            '+ellps=GRS80 +units=m +no_defs'
            )
        return Proj(_proj_str)

    # Note: if you need ITRF2000, use "epsg:4919"
    elif sys1 == 'WGS84' and sys2 == 'ITRF05':

        sys1 = '+init=epsg:{:04d}'.format(EPSG.WGS84.value)
        sys2 = '+init=epsg:{:04d}'.format(EPSG.ITRF05.value)
        return partial(transform, Proj(sys1), Proj(sys2))

    elif sys1 == 'ITRF05' and sys2 == 'WGS84':

        wgs84 = Proj('+proj=longlat +ellps=WGS84 +datum=WGS84 +no_defs')
        itrf05 = Proj('+init=epsg:4896')

        return partial(transform, itrf05, wgs84)

    elif sys1 == 'WGS84' and sys2 == 'ITRF08':

        wgs84 = Proj('+proj=longlat +ellps=WGS84 +datum=WGS84 +no_defs')
        itrf08 = Proj('+init=epsg:5332')

        return partial(transform, wgs84, itrf08)

    elif sys1 == 'ITRF08' and sys2 == 'WGS84':

        wgs84 = Proj('+proj=longlat +ellps=WGS84 +datum=WGS84 +no_defs')
        itrf08 = Proj('+init=epsg:5332')

        return partial(transform, itrf08, wgs84)

    else:
        raise AssertionError(
            'requested conversion not supported {} {}'.format(sys1, sys2)
            )

[docs]def utm_to_wgs84(ulon, ulat, zone, south=False):
    '''
    Convert UTM coordinates to GPS/WGS84.

    Parameters
    ----------
    ulon, ulat : `~astropy.units.Quantity`
        UTM longitude and latitude [m]
    zone : int
        UTM zone (e.g., 32 for longitudes 6 E to 12 E); see `Wikipedia
        <https://en.wikipedia.org/wiki/Universal_Transverse_Mercator_coordinate_system>`_
        for more information on the zones
    south : bool
        Hemisphere; set to True for southern hemisphere (default: False)

    Returns
    -------
    glon, glat : `~astropy.units.Quantity`
        GPS/WGS84 longitude and latitude [deg]

    Notes
    -----
    - Uses

      .. code-block:: bash

          +proj=utm +zone=xx +ellps=WGS84 +datum=WGS84 +units=m +no_defs [+south]

      for `pyproj` setup (inverse transformation). Only one zone per function
      call is allowed.
    - This function uses only the longitudal zone scheme. There is also
      the NATO system, which introduces latitude bands.
    '''

    _proj = _create_proj('UTM', 'WGS84', zone, south)

    return _proj(ulon, ulat, inverse=True)

    '''

    ----------
